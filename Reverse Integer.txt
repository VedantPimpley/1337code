class Solution:
    def reverse(self, x: int) -> int:
        '''
        requires:
            x is non-zero int
        ensures:
            -
        algo:
            if neg, set flag, set x to -1*x
            for each digit
                extract the digit
                write to output
            make output neg if x was neg
            return output
        '''

        # edge case
        if x == 0:
            return 0

        # general case
        num = x if x > 0 else -x
        exp = int(math.log(num, 10))
        
        res = 0
        while num > 0:
            d = num % 10
            num = num // 10
            res += d * math.pow(10, exp)
            exp -= 1
        res = -res if x < 0 else res
        res = int(res)

        tmp = math.pow(2,31)
        if res > tmp-1 or res < -tmp:
            return 0
        else:
            return res
